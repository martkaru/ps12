\subsection{Struktuurses operatsioonisemantikas}

\(
  \langle
  \text{while } x \neq y \text{ do }
  (\text{if } x \leq y
    \text{ then } y := y-x
    \text{ else } x:= x-y), s
  \rangle
  \\
  \quad \Rightarrow
  \langle
  \text{while } x \neq y \text{ do }
  (\text{ if } x \leq y
    \text{ then } y := y-x
    \text{ else } x:= x-y),
    s[y \mapsto 12]
  \rangle
\)
